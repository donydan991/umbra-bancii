on:
  schedule:
    - cron:  '*/15 * * * *'
  workflow_dispatch:
jobs:
  run-pipeline:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4
      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'
      - name: Install deps
       run: pip install -r requirements.txt
      - name: Run pipeline (demo)
        run: |
          python src/pipeline_demo.py
      - name: Upload report
        uses: actions/upload-artifact@v4
        with:
          name: reports
          path: output/
